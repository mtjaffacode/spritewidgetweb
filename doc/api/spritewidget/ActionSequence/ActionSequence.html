ActionSequence(List<Action> actions) {
  assert(actions.length >= 2);

  if (actions.length == 2) {
    // Base case
    _a = actions[0];
    _b = actions[1];
  } else {
    _a = actions[0];
    _b = new ActionSequence(actions.sublist(1));
  }

  // Calculate split and duration
  _duration = _a.duration + _b.duration;
  if (_duration > 0) {
    _split = _a.duration / _duration;
  } else {
    _split = 1.0;
  }
}
